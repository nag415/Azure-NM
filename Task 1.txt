script to create bootstrapped IIS server

Configuration WindowsFeatures
{
  param ([string[]]$NodeName = 'localhost')

  Node $NodeName
  {
    #Install the IIS Role
    WindowsFeature IIS
    {
      Ensure = “Present”
      Name = “Web-Server”
    }

  }
}
